## Key concepts learned

- **단일행 함수** - 행마다 적용되며 결과는 레코드 수만큼 반환되는 함수
- **그룹 함수** - 전체 또는 그룹 단위로 적용되며 결과는 하나 또는 그룹 수만큼 반환되는 함수
- **함수 중첩** - 여러 함수를 조합하여 사용할 수 있으며 안쪽부터 실행됨
- **CHAR_LENGTH vs LENGTH** - CHAR_LENGTH는 글자 수, LENGTH는 byte 수(한글은 3byte)
- **TRIM 함수의 변형** - BOTH/LEADING/TRAILING으로 특정 방향의 문자 제거 가능
- **GROUP BY 실행 순서** - WHERE(①) → GROUP BY(②) → HAVING(③) → SELECT(④) 순으로 실행
- **WHERE vs HAVING** - WHERE는 그룹 묶기 전 조건(그룹함수 불가), HAVING은 그룹 묶은 후 조건(그룹함수 가능)
- **CASE 문의 두 형태** - WHEN으로 동등비교하거나, WHEN으로 부등비교 가능
- **DATE_ADD/DATE_SUB** - INTERVAL 단위로 YEAR, MONTH, DAY 등을 더하거나 뺄 수 있음
- **TIMESTAMPDIFF** - 두 날짜 간 차이를 YEAR, MONTH, DAY, HOUR, MINUTE 등의 단위로 계산

## Commands / Configuration

### 문자열 함수

```sql
SELECT LOWER(컬럼명) FROM 테이블명;   -- 소문자로
SELECT UPPER(컬럼명) FROM 테이블명;   -- 대문자로

SELECT CONCAT(컬럼명1, 컬럼명2) FROM 테이블명;
SELECT CONCAT_WS(',', 컬럼명1, 컬럼명2) FROM 테이블명;

SELECT LPAD(컬럼명, 총길이, '채울문자') FROM 테이블명;  -- 왼쪽 채우기
SELECT RPAD(컬럼명, 총길이, '채울문자') FROM 테이블명;  -- 오른쪽 채우기

SELECT SUBSTR(컬럼명, 시작위치, 길이) FROM 테이블명;
SELECT LEFT(컬럼명, 길이) FROM 테이블명;   -- 왼쪽에서 N글자
SELECT RIGHT(컬럼명, 길이) FROM 테이블명;  -- 오른쪽에서 N글자

SELECT LENGTH(컬럼명) FROM 테이블명;       -- byte 수
SELECT CHAR_LENGTH(컬럼명) FROM 테이블명;  -- 글자 수

SELECT REPLACE(컬럼명, '바꿀값', '새값') FROM 테이블명;

SELECT INSTR(컬럼명, '찾을문자') FROM 테이블명;

SELECT LTRIM(컬럼명) FROM 테이블명;   -- 왼쪽 공백
SELECT RTRIM(컬럼명) FROM 테이블명;   -- 오른쪽 공백
SELECT TRIM(컬럼명) FROM 테이블명;    -- 양쪽 공백
SELECT TRIM(BOTH '삭제값' FROM 컬럼명) FROM 테이블명;     -- 양쪽 특정 문자
SELECT TRIM(LEADING '삭제값' FROM 컬럼명) FROM 테이블명;  -- 왼쪽 특정 문자
SELECT TRIM(TRAILING '삭제값' FROM 컬럼명) FROM 테이블명; -- 오른쪽 특정 문자

SELECT REVERSE(컬럼명) FROM 테이블명;        -- 거꾸로
SELECT REPEAT(컬럼명, 반복횟수) FROM 테이블명; -- 반복 출력
SELECT FORMAT(컬럼명, 소수점자릿수) FROM 테이블명; -- 천단위 콤마 포맷
```
