## Key concepts learned

- **DOM 처리 (DOM Manipulation)**: JS로 HTML 태그를 찾아서 값을 읽거나 수정하는 것
- **document.querySelector()**: CSS 선택자 문법으로 태그를 선택하는 메서드
- **innerHTML vs innerText vs value**: innerHTML은 태그를 포함한 내용, innerText는 텍스트만, value는 input 요소의 값을 다룸
- **이벤트 (Event)**: 사용자가 화면에서 취하는 동작 (클릭, 입력 등)
- **event.target**: 이벤트가 발생한 태그 자체에 대한 정보를 담은 객체
- **이벤트 연결 방식**: 인라인 방식과 init() 함수 방식의 차이
- **객체분해 할당 (Destructuring Assignment)**: 배열이나 JSON의 값을 간편하게 변수에 담는 ES6 문법
- **모듈 (Module)**: JS 코드를 여러 파일로 분리하고 다른 파일에서 가져다 쓰는 기능

### DOM 선택 및 조작

```javascript
// CSS 선택자로 태그 선택
document.querySelector("#id명");
document.querySelector(".class명");
document.querySelector("p");

// 태그의 상세 속성 확인
console.dir(태그);

// 내용 읽기/수정
p.innerHTML = "<h1>world</h1>";
p.innerText = "world";

// input 값 읽기
var input = document.querySelector("#username");

```

### DOM 처리 실습 예시 (계산기)

```javascript
function cal(op) {
    // 1. 태그에서 값 읽기
    var v1 = document.querySelector("#v1").value;
    var v2 = document.querySelector("#v2").value;

    // 2. 연산 (문자열 → 숫자 변환 필수!)
    var result;
    switch(op) {
        case "+": result = Number.parseInt(v1) + Number.parseInt(v2); break;
        case "*": result = Number.parseInt(v1) * Number.parseInt(v2); break;
    }

    // 3. 결과 화면에 출력
    document.querySelector("#result").innerHTML = result;
}
```

### 이벤트 연결 방식 (init() 함수 방식)

```javascript
function ok() {
    console.log("ok");
}

function init() {
    var btn = document.querySelector("#x");
    btn.onclick = ok;  // () 없이 함수 자체를 넘김
}
```

### 배열 분해할당

```javascript
// 기본
var [a, b] = [10, 20];
console.log(a, b);  // 10 20

// 기본값 설정
var [a, b, c=1, d=2] = [10, 20];
console.log(a, b, c, d);  // 10 20 1 2

// rest 파라미터
var [a, b, ...c] = [10, 20, 30, 40];
console.log(a, b, c);  // 10 20 [30, 40]

// 특정 위치 건너뛰기
var [, , c] = [10, 20, 30];
console.log(c);  // 30

// 변수 값 교환
var n = 10, m = 20;
[m, n] = [n, m];
console.log(n, m);  // 20 10
```

### JSON 분해할당

```javascript
// 기본
var {name, age} = {age: 20, name: '홍길동'};
console.log(name, age);  // 홍길동 20

// 기본값 설정
var {name, email="inky"} = {age: 20, name: '홍길동'};
console.log(name, email);  // 홍길동 inky

// 변수명 변경
var {age: x, name: y} = {age: 20, name: '홍길동'};
console.log(x, y);  // 20 홍길동

// 이미 선언된 변수에 할당 시 () 필수
var age, name;
({age, name} = {age: 20, name: '홍길동'});
console.log(name, age);  // 홍길동 20
```

### 모듈 내보내기 (export)

```javascript
// content.js
export function fun() {
    console.log("fun");
}

// header.js
export function fun2() {
    console.log("fun2");
}
export class Person {
    getMesg() { return "getMesg"; }
}

// body.js
export default function xxx() {  // default export: 파일당 1개만
    console.log("xxx");
}
export function yyy() { console.log("yyy"); }
export function zzz() { console.log("zzz"); }
```

### 모듈 가져오기 (import)

```javascript
// main.js
import {fun} from './content.js';
import {fun2, Person} from './header.js';
import xxx, {yyy, zzz} from './body.js';

// 별칭 지정 (as)
import {yyy as Y} from './body.js';
import {default as X} from './body.js';

fun();
fun2();
var p = new Person();
console.log(p.getMesg());
```
